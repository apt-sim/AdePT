if(NOT TARGET G4HepEm::g4HepEm)
  message(STATUS "Disabling example18 (needs G4HepEm)")
  return()
endif()

if(Geant4_FOUND)
  if(NOT Geant4_gdml_FOUND)
    message(STATUS "Disabling example18 (needs Geant4 with GDML support)")
    return()
  endif()
else()
  message(STATUS "Disabling example18 (needs Geant4)")
  return()
endif()

# example18 restricts the step length in case of magnetic field for better performance.
add_executable(example18
  example18.cpp
  example18.cu
  electrons.cu
  gammas.cu)
target_link_libraries(example18
  PRIVATE
    AdePT
    CopCore::CopCore
    VecGeom::vecgeom
    VecGeom::vecgeomcuda_static
    VecGeom::vgdml
    ${Geant4_LIBRARIES}
    G4HepEm::g4HepEmData
    G4HepEm::g4HepEmInit
    G4HepEm::g4HepEmRun
    CUDA::cudart)

set_target_properties(example18 PROPERTIES CUDA_SEPARABLE_COMPILATION ON CUDA_RESOLVE_DEVICE_SYMBOLS ON)

# NVTX
target_link_libraries(example18 PRIVATE NVTX)

# Tests
add_test(NAME example18
  COMMAND $<TARGET_FILE:example18> -gdml_file ${CMAKE_BINARY_DIR}/cms2018.gdml)
